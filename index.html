async function searchBooks(){ const q = document.getElementById("searchBox").value; if(!q){ document.getElementById("results").innerHTML = "Please enter a book title."; return; } document.getElementById("results").innerHTML = "Searching..."; const response = await fetch( "https://openlibrary.org/search.json?q=" + encodeURIComponent(q) ); const data = await response.json(); let html = '
'; for( let i = 0; i < Math.min(data.docs.length,12); i++ ){ const book = data.docs[i]; let cover; if(book.cover_i){ cover = `https://covers.openlibrary.org/b/id/${book.cover_i}-M.jpg`; }else{ cover = "https://via.placeholder.com/180x260?text=No+Cover"; } const author = book.author_name ? book.author_name[0] : "Unknown Author"; html += `

${book.title}

${author}

`; } html += "
"; document.getElementById("results").innerHTML = html; }
